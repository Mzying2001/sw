178 template <typename T, typename TField>
179 static FieldId GetFieldId(TField T::*field)
180 {
181 auto pfunc = &Reflection::GetFieldId<T, TField>;
182
183 uint8_t buffer[sizeof(pfunc) + sizeof(field)];
184 memcpy(buffer, &pfunc, sizeof(pfunc));
185 memcpy(buffer + sizeof(pfunc), &field, sizeof(field));
186
187 uint32_t prime = 16777619u;
188 uint32_t hash = 2166136261u;
189
190 for (size_t i = 0; i < sizeof(buffer); ++i) {
191 hash ^= static_cast<uint32_t>(buffer[i]);
192 hash *= prime;
193 }
194
195 return FieldId{hash};
196 }
